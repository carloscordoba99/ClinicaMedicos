CREATE PROCEDURE [dbo].[spAgregarPaciente]
(
    @DNI_P INT,
    @NOMBRE_P VARCHAR(30),
    @APELLIDO_P VARCHAR(30),
    @SEXO_P CHAR(1),
    @NACIONALIDAD_P VARCHAR(40),
    @FECHANAC_P DATE,
    @DIRECCION_P VARCHAR(40),
    @LOCALIDAD_P VARCHAR(30),
    @PROVINCIA_P VARCHAR(30),
    @EMAIL_P VARCHAR(30),
    @TELEFONO_P VARCHAR(30),
    @ESTADO_P BIT
)
AS
INSERT INTO Pacientes(
        DniPaciente_P, 
        Nombre_P, 
        Apellido_P, 
        Sexo_P, 
        Nacionalidad_P, 
        FechaNacimiento_P, 
        Direccion_P, 
        IdLocalidad_Paciente, 
        IdProvincia_Paciente, 
        Email_P, 
        Telefono_P, 
        Estado_P)
		Values(@DNI_P, 
        @NOMBRE_P, 
        @APELLIDO_P, 
        @SEXO_P, 
        @NACIONALIDAD_P, 
        @FECHANAC_P, 
        @DIRECCION_P, 
        @LOCALIDAD_P, 
        @PROVINCIA_P, 
        @EMAIL_P, 
        @TELEFONO_P, 
        @ESTADO_P)
	RETURN
